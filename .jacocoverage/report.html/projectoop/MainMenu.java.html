    private void utpButton_MenuActionPerformed(java.awt.event.ActionEvent evt) {//GEN-FIRST:event_utpButton_MenuActionPerformed
        try{
                String QueryUtp = "select * from account where id_account="+IdUser+" and id_type=1";
                Con = DriverManager.getConnection("jdbc:mysql://localhost:3306/primodb-test","root","");
                St = Con.createStatement();
                Rs = St.executeQuery(QueryUtp);
                if(Rs.next()){  
                    new UTPMenu(IdUser).setVisible(true);
                    this.dispose();
                } else {
                    JOptionPane.showMessageDialog(this, "Account is Already Premium");
                }
            }catch(Exception e){
                JOptionPane.showMessageDialog(this, e);
            }
    }//GEN-LAST:event_utpButton_MenuActionPerformed

    private void transferBalanceButton_MenuMouseClicked(java.awt.event.MouseEvent evt) {//GEN-FIRST:event_transferBalanceButton_MenuMouseClicked
            try{
                String QueryAcc = "select * from account where id_account="+IdUser+" and id_type=2";
                Con = DriverManager.getConnection("jdbc:mysql://localhost:3306/primodb-test","root","");
                St = Con.createStatement();
                Rs = St.executeQuery(QueryAcc);
                if(Rs.next()){  
                    new TransferBalanceMenu(IdUser).setVisible(true);
                    this.dispose();
                } else {
                    JOptionPane.showMessageDialog(this, "Account is Not Premium");
                }
            }catch(Exception e){
                JOptionPane.showMessageDialog(this, e);
            }
    }//GEN-LAST:event_transferBalanceButton_MenuMouseClicked
